  (print-second sec)
  (let ((new-sec (+1 sec)))
    (if (>=60? new-sec)
        ()  
        (synchronize (count-seconds new-sec)))))

(defun count-minutes (min)
  (print-minute min)
  (count-seconds 0)
  (let ((new-min (+1 min)))
    (if (>=60? new-min)
        ()
